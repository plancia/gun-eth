üîê Gun-Ethereum Integration

Authentication Example


// Initialize Gun-Eth
import { GunEth } from 'gun-eth';
import Gun from 'gun';

// Configura Gun con i peers
const gun = Gun({
    peers: ['http://localhost:8765/gun']
});

// Inizializza GunEth
const gunEth = new GunEth(gun);

// Converti un account Ethereum in un account Gun
import { ethToGunAccount, createSignature, MESSAGE_TO_SIGN } from 'gun-eth';

// Crea una firma con il wallet Ethereum
const signature = await createSignature(MESSAGE_TO_SIGN);

// Converti in account Gun
const gunAccount = await ethToGunAccount(signature);

// L'account contiene:
console.log({
    pub: gunAccount.pub,    // Chiave pubblica Gun
    priv: gunAccount.priv,  // Chiave privata Gun
    epub: gunAccount.epub,  // Chiave pubblica di cifratura
    epriv: gunAccount.epriv // Chiave privata di cifratura
});

// Converti un account Gun in un account Ethereum
import { gunToEthAccount } from 'gun-eth';

// Usa un account Gun esistente
const gunAccount = {
    pub: "chiave-pubblica-gun",
    priv: "chiave-privata-gun",
    epub: "chiave-pubblica-cifratura",
    epriv: "chiave-privata-cifratura"
};

// Converti in account Ethereum
const ethAccount = await gunToEthAccount(gunAccount);

// L'account contiene:
console.log({
    address: ethAccount.address,     // Indirizzo Ethereum
    privateKey: ethAccount.privateKey // Chiave privata Ethereum
});

// Esempio di utilizzo completo
const signature = await createSignature(MESSAGE_TO_SIGN);
const gunAccount = await ethToGunAccount(signature);

// Salva dati cifrati
await gun.user(gunAccount.pub).get('myData').put({
    secret: "Dati cifrati con le chiavi Gun"
});

// Recupera dati cifrati
const data = await gun.user(gunAccount.pub).get('myData').then();

// Converti indietro in Ethereum se necessario
const ethAccount = await gunToEthAccount(gunAccount);

Initial Setup


// Initialize StealthChain
import { StealthChain } from 'gun-eth';
import Gun from 'gun';

const gun = Gun({
    peers: ['your-gun-peer']
});

const stealthChain = new StealthChain(gun);

// Create and publish stealth keys
const signature = await createSignature(MESSAGE_TO_SIGN);
await stealthChain.publishStealthKeys(signature);

// Verify key publication
await waitForGunData(
    gun,
    'gun-eth/users/' + userAddress + '/publicKeys',
    data => data?.viewingPublicKey && data?.spendingPublicKey
);

// Generate stealth address
const stealthInfo = await stealthChain.generateStealthAddress(
    recipientAddress,
    senderSignature
);

// Announce payment
await stealthChain.announceStealthPayment(
    stealthInfo.stealthAddress,
    stealthInfo.senderPublicKey,
    stealthInfo.spendingPublicKey,
    senderSignature,
    { chain, onChain: true }
);

‚õìÔ∏è Proof Chain

Data Verification


// Initialize ProofChain
import { ProofChain } from 'gun-eth';
import Gun from 'gun';

const gun = Gun(options);
const proofChain = new ProofChain(gun);

// Write data with proof
proofChain.proof(
    chain,
    null,  // nodeId (null for new data)
    data,
    (ack) => {
        if (ack.ok) {
            console.log("Node ID:", ack.nodeId);
            console.log("Transaction Hash:", ack.txHash);
        }
    }
);

// Verify data integrity
gun.proof(
    chain,
    nodeId,
    null,
    (ack) => {
        if (ack.ok) {
            console.log("Verification passed");
            console.log("Last update:", ack.timestamp);
            console.log("Updater:", ack.updater);
        }
    }
);

// Monitor data changes
gun.get(nodeId).on((data, key) => {
    // Check data integrity
    const originalHash = data._contentHash;
    const currentHash = calculateHash(data);
    
    if (currentHash !== originalHash) {
        console.warn("Data tampering detected!");
        console.log("Original hash:", originalHash);
        console.log("Current hash:", currentHash);
    }
});

// Initialize Bubble Client
import { BubbleClient } from 'gun-eth';

// Create client instance
const bubbleClient = new BubbleClient({
    providerUrl: "http://localhost:3000/api",
    signer: signer,  // Your Ethereum signer
    keypair: {       // Your encryption keypair
        epub: "your-encryption-public-key",
        epriv: "your-encryption-private-key"
    }
});

// Create a new bubble
const bubble = await bubbleClient.createBubble("My Bubble", {
    isPrivate: true
});

// Write encrypted content
await bubbleClient.writeBubble(
    bubble.id,
    "secret.txt",
    "My secret content"
);

// Read and decrypt content

    bubble.id,
    "secret.txt"
);

// Share with another user
await bubbleClient.shareBubble(
    bubble.id,
    recipientAddress,
    { granteeEpub: recipientPublicKey }
);

Provider Types


// Base Provider - Abstract class for all providers
import { BaseBubbleProvider } from 'gun-eth';

class CustomProvider extends BaseBubbleProvider {
    // Required methods to implement
    async handleCreateBubble(options) { }
    async handleWriteBubble(bubbleId, fileName, content) { }
    async handleReadBubble(bubbleId, fileName) { }
    async handleGrantPermission(bubbleId, granteeAddress) { }
}

// GUN Provider - Uses GUN for storage
import { GUNBubbleProvider } from 'gun-eth';

const gunProvider = new GUNBubbleProvider({
    gun: gunInstance,
    signer: signer,
    keypair: keypair
});

// Stores data directly in GUN's graph
await gunProvider.handleCreateBubble({
    name: "GUN Bubble",
    isPrivate: true
});

// Hybrid Provider - Combines GUN with external storage
import { HybridBubbleProvider } from 'gun-eth';

const hybridProvider = new HybridBubbleProvider({
    gun: gunInstance,
    signer: signer,
    keypair: keypair,
    storage: customStorageAdapter
});

// Supports both GUN and external storage
await hybridProvider.handleCreateBubble({
    name: "Hybrid Bubble",
    isPrivate: true,
    useExternal: true
});
